    # 一、button 按钮

    # 支持属性

    • type string ('primary','warning','danger','success','info') ----按钮类型
    • icon string ('icon-系列',作者正在补充当中) ----字体图标
    • iconPosition string ('left','right') ----用于控制 icon 的显示位置
    • loading boolean ('true','false') ----加载中

    # 支持事件

    • click 事件

    # 举个栗子

    <little-button>默认按钮</little-button>
<little-button type="primary" icon="icon-wode" iconPosition="right">
  主要按钮
</little-button>
<little-button type="info" icon="icon-shezhi">提示按钮</little-button>
<little-button type="danger" icon="icon-jinru">危险按钮</little-button>
<little-button type="warning" icon="icon-gengduo2" loading>
  警示按钮
</little-button>
<little-button type="success" icon="icon-rizhi" @click="btn">
  成功按钮可点击
</little-button>
<little-button-group>
  <little-button icon="icon-jinru">进入</little-button>
  <little-button icon="icon-shanchu">删除</little-button>
</little-button-group>

    # 三、card 卡片

    # 支持属性

    • width number （正整数）----卡片宽度
    • height number （正整数）----卡片高度
    • descColor string （支持 rgb,16 进制,内置颜色）----描述背景色
    • descOpacity number （0-1）----描述透明度
    • descHeight number （正整数）----描述盒子高度

    # 支持事件

    • click 事件

    # 插槽

    • 具名插槽 "image" 卡片整体
    • 具名插槽 "desc" 卡片描述

    # 举个栗子

      <little-card :descOpacity="0.7" @click="clickCard">
    <div
    slot="image"
    class="image"
    :style="{
      backgroundImage: `url(${cardImage})`,
      backgroundSize: 'cover',
      backgroundPosition: 'center',
    }"
    >
    </div>
    <div slot="desc">this is descthis is descthis is desc</div>
  </little-card>

    # 四、dialog 弹窗

    # 支持属性

    • visible boolean（‘true’, 'false'）----控制弹层显示与否
    • showClose boolean（'true','false'）----控制关闭按钮显示与否
    • width number（正整数）----自定义弹窗宽度，附，会覆盖 h5 端适配样式

    # 支持事件

    • 正在补充中

    # 插槽

    • 具名插槽 "title" 弹窗标题
    • 具名插槽 "foot" 弹窗页脚
    • 默认插槽 弹窗内容

    # 举个栗子

      <template>
    <little-dialog :visible.sync="secondVisible" :showClose="false">
      <div
        class="dialog-image"
        :style="{
          backgroundImage: `url(${cardImage})`,
          backgroundSize: 'cover',
          backgroundPosition: 'center',
        }"
      ></div>
      <div slot="foot">
        <little-button type="primary" @click="secondVisible = false">
          确定
        </little-button>
        <little-button type="info" @click="secondVisible = false">
          取消
        </little-button>
      </div>
    </little-dialog>
  </template>

  <script>
  export default{
    data() {
      return {
        cardImage: require("./image/avatar.jpg"),
        secondVisible: false,
      };
    },
    methods: {
    }
  }
  </script>
